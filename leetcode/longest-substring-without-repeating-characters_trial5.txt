class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        left = 0
        window = set()
        longest = 0
        for i in range(len(s)):
            while left < len(s) and s[i] in window:
                window.remove(s[left])

                left += 1
            window.add(s[i])
            longest = max(longest,len(window))
        return longest
