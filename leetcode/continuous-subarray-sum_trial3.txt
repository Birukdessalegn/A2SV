class Solution:
    def checkSubarraySum(self, nums: List[int], k: int) -> bool:
        total = 0
        seen = {0: -1} 

        for i in range(len(nums)):
            total += nums[i]
            rem = total % k if k != 0 else total

            if rem in seen:
                if i - seen[rem] >= 2: 
                    return True
            else:
                seen[rem] = i 

        return False
