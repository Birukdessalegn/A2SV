class Solution:
    def maxScore(self, cardPoints: List[int], k: int) -> int:
        res = 0
        total = 0
        n = len(cardPoints)
        total = sum(cardPoints[:k])
        res = total
        for i in range(1,k+1):
            total -= cardPoints[k-i]

            total += cardPoints[n-i]
            res = max(res,total)
        if k == 1:
            res = max(cardPoints[0],cardPoints[-1])
        if n == k:
            res = sum(cardPoints)
        return res
