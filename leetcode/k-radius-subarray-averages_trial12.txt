class Solution:
    def getAverages(self, nums: List[int], k: int) -> List[int]:
        n = len(nums)
        avgs = [-1] * n
        
        window_size = 2 * k + 1

        if window_size > n:
            return avgs
        
        window_sum = sum(nums[:window_size])

        avgs[k] = window_sum // window_size

        for i in range(window_size, n):
            window_sum += nums[i]             
            window_sum -= nums[i - window_size]  
            center = i - k
            avgs[center] = window_sum // window_size
        
        return avgs
