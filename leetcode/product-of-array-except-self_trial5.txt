class Solution:
    def productExceptSelf(self, nums: List[int]) -> List[int]:
        ans = []
        zero = nums.count(0)
        total = prod(x for x in nums if x != 0)
        for i in range(len(nums)):
            if zero > 1:
                res = 0
            elif zero == 1:
                if nums[i] == 0:
                    res = total
                else:
                    res = 0
            else:
                res = total//nums[i]
            ans.append(res) 
        return ans
