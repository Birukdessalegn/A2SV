class Solution:
    def totalFruit(self, fruits: List[int]) -> int:
        count= Counter()
        left = 0
        res = 0
        for right in range(len(fruits)):
            count[fruits[right]] += 1
            
            while len(count) > 2:
                left_fruit = fruits[left]
                count[left_fruit] -= 1
                if count[left_fruit] == 0:
                    del count[left_fruit]
                left += 1
            res = max(res , right-left+1)
        return res
